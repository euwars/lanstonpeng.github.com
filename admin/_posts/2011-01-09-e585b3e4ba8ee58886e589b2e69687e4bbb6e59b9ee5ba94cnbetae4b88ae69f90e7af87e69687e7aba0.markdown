split -b20M android.pdf android_part_*( 每20M分割,还是显然看不到的)

用ghostscript合并pdf文件:     gs -dBATCH -dNOPAUSE -q -sDEVICE=pdfwrite -sOutputFile=merged_file.pdf pdf_file_1.pdf pdf_file_2.pdf
